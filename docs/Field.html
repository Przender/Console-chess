def to_s
  if @piece
    bg = @color == :black ? 40 : 47
    fg = @piece.color == :black ? 34 : 33   #30 : 37

    "\e[#{bg}m\e[#{fg}m#{@piece.to_s}\e[0m"
  else
    bg = @color == :black ? 40 : 47
    "\e[#{bg}m \e[0m"
  end
end

def update(new_piece)
  if @piece && new_piece
    Chess.players[@piece.color].delete_piece(@piece)
  end

  @piece = new_piece

  @piece.pos = @pos if @piece
end
