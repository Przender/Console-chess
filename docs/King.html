def in_check
  enemy_color = (color == :white ? :black : :white)
  is_checked = false

  Chess.players[enemy_color].pieces.values.flatten.each do |piece|
    piece.get_possible_moves
    if piece.poss_moves.include?([@pos[0], pos[1], :attacking])
      is_checked = true
      @checked_by << piece unless @checked_by.find_index(piece)
    end
    piece.turn_reset
  end

  is_checked
end

    true if the king is currently in check by checking if the field it stands on is being attacked by any of the opponent’s pieces
